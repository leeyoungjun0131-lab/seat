function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Reservations'); const data = JSON.parse(e.postData.contents); if(data.action === 'reserve') { sheet.appendRow([data.date, data.seatId, data.userName, data.userId]); } else if(data.action === 'cancel') { const rows = sheet.getDataRange().getValues(); for(let i = rows.length - 1; i >= 1; i--) { if(rows[i][0] === data.date && rows[i][1] === data.seatId) { sheet.deleteRow(i + 1); break; } } } return ContentService.createTextOutput(JSON.stringify({result:'OK'})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Reservations'); const today = new Date().toISOString().slice(0,10); const rows = sheet.getDataRange().getValues(); const result = {}; for(let i = 1; i < rows.length; i++) { if(rows[i][0] === today) { result[rows[i][1]] = { userId: rows[i][3].replace('@promega.com',''), userName: rows[i][2] }; } } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
